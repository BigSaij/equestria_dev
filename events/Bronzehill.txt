	option = { #Oh No
		name = bronzehill_revenge.1.a

			}
		}
		leave_faction = yes
		add_opinion_modifier = {
			target = GRI
			modifier = sworn_enemies
		}
		add_opinion_modifier = {
			target = GRI
			modifier = BRZ_murdered_the_emperor
		}
		effect_tooltip = {
			load_focus_tree = brz_revenge_tree
		}

		hidden_effect = {
			set_country_flag = herzland_diplomacy_blocked
			add_ai_strategy = {
				type = befriend
				id = GRI
				value = -200
			}
			add_ai_strategy = {
				type = protect
				id = GRI
				value = -200
			}
			add_ai_strategy = {
				type = send_volunteers_desire
				id = GRI
				value = -200
			}
			add_ai_strategy = {
				type = support
				id = GRI
				value = -200
			}
			add_ai_strategy = {
				type = alliance

				value = -200
