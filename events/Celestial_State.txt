	option = {
		name = daring_harmony.1.a

			remove_idea = CES_no_real_government_0

		}
		if = {
		    limit = { country_exists = EQS }
		    EQS = { 
                puppet = NMS			
		    }
			EQS = { 
		        add_to_faction = NMS 
		    }
	    }
		if = {
		    limit = { 
			    country_exists = NLR
                NLR = { has_war_with = EQS }				
			}
			hidden_effect = {
				create_wargoal = { 
					type = annex_everything
					target = NLR
				}
			}
		    declare_war_on = {
                target = NLR
                type = annex_everything
            }
	    }
	}
}
