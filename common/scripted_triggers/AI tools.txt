ai_is_doing_border_war_as_THIS_with_FROM_appropriate = {
    OR = {
        # generic checks to check if we are not too likely in need of more troops somewhere else
        AND = {
            NOT = {
                has_war = yes
            }
            all_neighbor_country = {
                PREV = {
                    check_variable = {
                        ai_strategy_prepare_for_war@PREV < 100
                    }
                }
            }
            num_divisions > 7
        }
        # circumvent previous checks if we expect to be at war with the nation soon anyway
        check_variable = {
            ai_strategy_prepare_for_war@FROM > 49
        }
        is_justifying_wargoal_against = FROM

            is_justifying_wargoal_against = PREV
        }
    }
}
