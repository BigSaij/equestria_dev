scripted_gui = {
	ABY_Parliament = {
		context_type = decision_category
		window_name = ABY_Parliament_GUI
		dirty = ABY_PLMT_dirty
		
		dynamic_lists = {
			parliament_glow_gridbox = {
				array = ABY_parliament_x
				entry_container = ABY_parliament_diagram_glow_entry
				index = seat_idx
			}
			parliament_diagram = {
				array = ABY_parliament_x
				entry_container = ABY_parliament_diagram_entry
				index = seat_idx
			}
		}

		properties = {
			parliament_dot_icon = {
				x = ABY_parliament_x^seat_idx
				y = ABY_parliament_y^seat_idx
				frame = ABY_parliament_seats_array_frame^seat_idx
			}
			parliament_glow = {
				x = ABY_parliament_x^seat_idx
				y = ABY_parliament_y^seat_idx
			}
		}
		
		triggers = {
			parliament_glow_visible = {
				OR = {
					AND = {
						has_country_flag = ABY_urf_split
						OR = {	
							is_in_array = { ABY_coalitions_array = ABY_parliament_seats_array_display^seat_idx }
							check_variable = { ABY_parliament_seats_array_display^seat_idx = 2 }
						}
					}
					AND = {
						NOT = {
							has_country_flag = ABY_urf_split
						}
						OR = {
							check_variable = { ABY_PLMT_nosplit_additional_array^seat_idx = 1 }
							check_variable = { ABY_parliament_seats_array_display^seat_idx = 0 }
						}
					}
				}
			}
		
			URF_seats_visible = {
				NOT = { has_country_flag = ABY_urf_split }
			}
			APF_seats_1_visible = {
				NOT = { has_country_flag = ABY_urf_split }
			}
			
			ASP_seats_visible = {
				has_country_flag = ABY_urf_split
			}
			ALF_seats_visible = {
				has_country_flag = ABY_urf_split
			}
			PLA_seats_visible = {
				has_country_flag = ABY_urf_split
			}
			APF_seats_2_visible = {
				has_country_flag = ABY_urf_split
			}
		}
	}
}
