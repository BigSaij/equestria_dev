defined_text = {
	# def text to get the name of HCH character. Inputs ids and values in hch_value
	name = GetHCHname
	text = {
		trigger = {
			set_temp_variable = { id_to_check = 0.001 }
			is_same_id_for_value_trigger = yes
		}
		localization_key = kir_hch_1_name
	}
	text = {
		trigger = {
			set_temp_variable = { id_to_check = 0.002 }
			is_same_id_for_value_trigger = yes
		}
		localization_key = kir_hch_2_name
	}
	text = {
		trigger = {
			set_temp_variable = { id_to_check = 0.003 }
			is_same_id_for_value_trigger = yes
		}
		localization_key = kir_hch_3_name
	}
}

defined_text = {
	# def text to get the character cost
	name = GetHCHcost
	text = {
		trigger = {
			check_variable = { selected_hch_level = 1 }
			has_political_power > 49
		}
		localization_key = kir_hierarchy_level_1_cost
	}
	text = {
		trigger = {
			check_variable = { selected_hch_level = 1 }
			NOT = { has_political_power > 49 }
		}
		localization_key = kir_hierarchy_level_1_cost_blocked
	}
	text = {
		trigger = {
			check_variable = { selected_hch_level = 2 }
			has_political_power > 24
		}
		localization_key = kir_hierarchy_level_2_cost
	}
	text = {
		trigger = {
			check_variable = { selected_hch_level = 2 }
			NOT = { has_political_power > 24 }
		}
		localization_key = kir_hierarchy_level_2_cost_blocked
	}
	text = {
		trigger = {
			OR = {
				check_variable = { selected_hch_level = 3 }
				check_variable = { selected_hch_level = 4 }
				check_variable = { selected_hch_level = 5 }
			}
			has_political_power > 9
		}
		localization_key = kir_hierarchy_level_3_cost
	}
	text = {
		trigger = {
			OR = {
				check_variable = { selected_hch_level = 3 }
				check_variable = { selected_hch_level = 4 }
				check_variable = { selected_hch_level = 5 }
			}
			NOT = { has_political_power > 9 }
		}
		localization_key = kir_hierarchy_level_3_cost_blocked
	}
}
