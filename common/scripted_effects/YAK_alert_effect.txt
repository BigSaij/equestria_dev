### IN ORDER: ###

# mankhutayid - 5
# suldintjid - 6
# yurkid - 7
YAK_alert_raise_juyilkhai = {
	if = {
		limit = { is_ai = no }
		#Creates an alert about the Juyilkhai clan
		add_to_array = { YAK_alert_type_array = 1 }
		add_to_array = { YAK_alert_clan_array = 0 }
		sound_effect = "alert_free_civilian_factories_sound"
	}
}
YAK_alert_remove_juyilkhai = {
	if = {
		limit = { is_ai = no }
		#Removes the alert about the Juyilkhai clan
		set_temp_variable = { value_to_remove = 0 }
		for_each_loop = {
			array = YAK_alert_clan_array
			value = v
			index = i
			break = break
			if = {
				limit = {
					check_variable = {
						v = 0
					}
				}
				set_temp_variable = { value_to_remove = i }
				set_temp_variable = { break = 1 }
			}
		}
		hidden_effect = {
			remove_from_array = { array = YAK_alert_type_array index = value_to_remove }
			remove_from_array = { array = YAK_alert_clan_array index = value_to_remove }
		}
	}
}
