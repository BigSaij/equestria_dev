### THE NUMEBRS ARE DIFFERENT BECAUSE OF FRAME WEIRDNESS AND THE FACT GHENIKHAN CAN'T SECEDE ###
# juyilkhai - 1
# noyamanan - 2
# arusharii - 3
# kiyatukh - 4
# mankhutayid - 5
# suldintjid - 6
YAK_alert_raise_juyilkhai = {
	if = {
		limit = { is_ai = no }
		#Creates an alert about the Juyilkhai clan
		add_to_array = { YAK_alert_clan_array = 1 }
		sound_effect = "alert_free_civilian_factories_sound"
	}
}
YAK_alert_remove_juyilkhai = {
	if = {
		limit = { is_ai = no }
		#Removes the alert about the Juyilkhai clan
		set_temp_variable = { value_to_remove = -1 } # not 0 because 0 is the first value of the array
		for_each_loop = {
			array = YAK_alert_clan_array
			value = v
			index = i
			break = break
			if = {
				limit = {
					check_variable = {
						v = 1
					}
				}
				set_temp_variable = { value_to_remove = i }
				set_temp_variable = { break = 1 }
			}
		}
		hidden_effect = {
			if = { # required because otherwise it might erroneously remove the first value
				limit = {
					check_variable = {
						value_to_remove > -1
					}
				}
				remove_from_array = { array = YAK_alert_clan_array index = value_to_remove }
			}
		}
	}
}
